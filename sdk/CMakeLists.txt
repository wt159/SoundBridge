cmake_minimum_required(VERSION 3.2)

project(sdk)

set(3RDPARTY_DIST_DIR ${SDK_ROOT_DIR}/3rdparty/dist/${CMAKE_TARGET_SYSTEM_NAME}_${CMAKE_TARGET_SYSTEM_PROCESSOR}_${CMAKE_TARGET_TOOLCHAIN_NAME})
